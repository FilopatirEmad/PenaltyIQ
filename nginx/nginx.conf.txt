# PenaltyIQ Nginx Configuration
# Reverse proxy for FastAPI backend with upload size limit

worker_processes auto;

events {
    worker_connections 1024;
}

http {
    # ── Upload size: 500MB for video files ─────────────────────────────────
    client_max_body_size 500M;
    client_body_timeout  300s;
    proxy_read_timeout   300s;
    proxy_send_timeout   300s;

    upstream penaltyiq_backend {
        server penaltyiq-api:8000;
        keepalive 32;
    }

    server {
        listen 80;
        server_name _;

        # Redirect HTTP → HTTPS in production
        # return 301 https://$host$request_uri;

        location / {
            proxy_pass         http://penaltyiq_backend;
            proxy_http_version 1.1;
            proxy_set_header   Connection "";
            proxy_set_header   Host $host;
            proxy_set_header   X-Real-IP $remote_addr;
            proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header   X-Forwarded-Proto $scheme;

            # WebSocket support (for future real-time features)
            proxy_set_header   Upgrade $http_upgrade;
        }

        # Health check endpoint (no auth required)
        location /health {
            proxy_pass http://penaltyiq_backend/health;
            access_log off;
        }
    }
}
